#!/bin/bash

ADDRESS=`cat /dev/stdin | ~/bounce_address.py`

MESSAGEID=`tmda-pending -b -T | grep -i "$ADDRESS" | cut -f 1`

if [ "$MESSAGEID" != "" ] ;  then
    tmda-pending -b -d $MESSAGEID
else
    echo Message from $ADDRESS not found pending
fi
