#!/bin/sh

FILE=$1
CONFIRMADDR=`cat $FILE | grep '^Reply-To:.*-confirm-' | cut -f2 -d' '`

if [ -z "$CONFIRMADDR" ]; then
        exit
fi

MSGID=`echo $CONFIRMADDR | sed 's/-confirm-/:/' | cut -f2 -d: | \
        cut -f1-2 -d.`

tmda-pending -qbd $MSGID
